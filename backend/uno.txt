// ─────────────────────────────────────────────────────────────────────
// HELPERS
// ─────────────────────────────────────────────────────────────────────
float smoothSpeed(float v) {
  speedBuf[speedIdx] = v;
  speedIdx = (speedIdx + 1) % 5;
  float s = 0;
  for (int i = 0; i < 5; i++) s += speedBuf[i];
  return s / 5.0;
}

// ─────────────────────────────────────────────────────────────────────
// DHT22
// ─────────────────────────────────────────────────────────────────────
void readDHT() {
  float h = dht.readHumidity();
  float t = dht.readTemperature();
  if (isnan(h) || isnan(t)) {
    dhtValid = false;
    Serial.println("DHT22: Read failed — check GPIO4");
    return;
  }
  cTemp    = t;
  cHum     = h;
  cHeatIdx = dht.computeHeatIndex(t, h, false);
  dhtValid = true;
  Serial.printf("DHT22: %.1fC  %.1f%%  HI:%.1fC\n", t, h, cHeatIdx);
}

// ─────────────────────────────────────────────────────────────────────
// MPU6050_light — sample and detect rash events
// MPU6050_light already handles:
//   - Calibration offset removal
//   - Converting raw → m/s² and deg/s
//   - Complementary filter for pitch/roll angles
// ─────────────────────────────────────────────────────────────────────
void sampleMPU() {
  if (!mpuReady) return;

  // Update internal filter — must call this regularly
  mpu.update();

  // Get linear acceleration in m/s²
  ax = mpu.getAccX();
  ay = mpu.getAccY();
  az = mpu.getAccZ();

  // Get rotation rate in deg/s
  gx = mpu.getGyroX();
  gy = mpu.getGyroY();
  gz = mpu.getGyroZ();

  // Get filtered tilt angles (complementary filter inside library)
  pitch = mpu.getAngleX();
  roll  = mpu.getAngleY();

  // Total acceleration magnitude
  aMag = sqrt(ax*ax + ay*ay + az*az);

  // Net Z (remove gravity component ~9.81)
  float azNet  = abs(az) - 9.81;
  // Lateral + forward magnitude
  float xyMag  = sqrt(ax*ax + ay*ay);

  // ── Rash driving detection ───────────────────────────────────────
  // Harsh braking: strong negative X axis (forward deceleration)
  if (ax < RASH_BRAKE_MS2) {
    rashBrake = true;
    Serial.println("RASH: HARSH BRAKE");
  }

  // Harsh acceleration: strong positive X axis
  if (ax > RASH_ACCEL_MS2) {
    rashAccel = true;
    Serial.println("RASH: HARSH ACCEL");
  }

  // Harsh turn: high lateral Y OR high gyro Z rotation rate
  if (abs(ay) > RASH_TURN_MS2 || abs(gz) > RASH_GYRO_DEG) {
    rashTurn = true;
    Serial.println("RASH: SHARP TURN");
  }

  // Bump/pothole: Z spike well above gravity
  if (azNet > RASH_BUMP_MS2) {
    rashBump = true;
    Serial.println("RASH: BUMP/POTHOLE");
  }
}

// ─────────────────────────────────────────────────────────────────────
// OBSTACLE — IR sensor (Option A)
// Digital read: LOW = obstacle present on most modules
// ─────────────────────────────────────────────────────────────────────
void readIR() {
  // Take 5 readings over 50ms — require at least 4/5 to confirm obstacle
  // This prevents false triggers from electrical noise
  int detCount = 0;
  for (int i = 0; i < 5; i++) {
    if (digitalRead(IR_PIN) == IR_OBSTACLE_ACTIVE) detCount++;
    delay(10);
  }
  obstDetected = (detCount >= 4);          // 4 of 5 readings must agree
  obstDist     = obstDetected ? 20.0 : 999.0; // 999 = clear, no obstacle
  Serial.printf("IR: %s (count=%d/5)\n", obstDetected ? "OBSTACLE" : "Clear", detCount);
}

// ─────────────────────────────────────────────────────────────────────
// OBSTACLE — HC-SR04 ultrasonic (Option B)
// ECHO pin must use voltage divider: 5V→1kΩ→GPIO14→2kΩ→GND
// ─────────────────────────────────────────────────────────────────────
void readUltrasonic() {
  digitalWrite(TRIG_PIN, LOW);  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH); delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);

  long duration = pulseIn(ECHO_PIN, HIGH, 30000); // 30ms timeout
  if (duration == 0) {
    obstDist     = 400.0;   // Out of range
    obstDetected = false;
  } else {
    obstDist     = (duration * 0.0343) / 2.0;
    obstDist     = constrain(obstDist, 2.0, 400.0);
    obstDetected = (obstDist < ULTRASONIC_WARN_CM);
  }
  Serial.printf("HC-SR04: %.1fcm %s\n", obstDist, obstDetected ? "[!]" : "OK");
}

void readObstacle() {
  if (String(OBSTACLE_SENSOR) == "ULTRASONIC") readUltrasonic();
  else                                          readIR();
}

// ─────────────────────────────────────────────────────────────────────
// SOCKET.IO
// ─────────────────────────────────────────────────────────────────────
void sendAuth() {
  StaticJsonDocument<256> d;
  d["token"] = JWT_TOKEN;
  String s; serializeJson(d, s);
  wsClient.sendTXT("40" + s);
}

void webSocketEvent(WStype_t type, uint8_t* payload, size_t length) {
  switch (type) {

    case WStype_DISCONNECTED:
      wsConnected = false;
      Serial.println("WS: Disconnected");
      break;

    case WStype_CONNECTED:
      wsConnected = true;
      Serial.println("WS: Connected");
      sendAuth();
      break;

    case WStype_TEXT: {
      String msg = String((char*)payload);
      if (msg == "3") { wsClient.sendTXT("2"); break; } // pong

      if (msg.startsWith("42")) {
        StaticJsonDocument<512> doc;
        deserializeJson(doc, msg.substring(2));
        String ev = doc[0].as<String>();

        if (ev == "error") {
          Serial.println("Server: " + doc[1]["message"].as<String>());
        }
        if (ev == "emergencyAlertSent") {
          Serial.println("Emergency confirmed");
        }
        if (ev == "safetyScore") {
          serverScore = doc[1]["score"].as<float>();
          serverBand  = doc[1]["band"].as<String>();
          Serial.printf("Score: %.1f/10 [%s]\n", serverScore, serverBand.c_str());
        }
      }
      break;
    }

    case WStype_ERROR:
      Serial.println("WS: Error");
      break;
  }
}

// ─────────────────────────────────────────────────────────────────────
// EMIT TELEMETRY — every 2s, driver only
// ─────────────────────────────────────────────────────────────────────
void emitTelemetry() {
  if (!wsConnected || String(DEVICE_ROLE) != "driver") return;

  float lat = gps.location.isValid() ? gps.location.lat() : 0.0;
  float lng = gps.location.isValid() ? gps.location.lng() : 0.0;
  float spd = smoothSpeed(gps.speed.isValid() ? gps.speed.kmph() : 0.0);
  float hdg = gps.course.isValid() ? gps.course.deg() : 0.0;

  StaticJsonDocument<768> doc;

  doc["busId"]     = BUS_ID;

  // GPS
  doc["latitude"]   = lat;
  doc["longitude"]  = lng;
  doc["speed"]      = spd;
  doc["heading"]    = hdg;
  doc["gpsFix"]     = gps.location.isValid();
  doc["satellites"] = gps.satellites.isValid() ? gps.satellites.value() : 0;
  doc["hdop"]       = gps.hdop.isValid() ? gps.hdop.hdop() : 99.9;

  // Obstacle
  doc["obstacleDistance"] = obstDist;
  doc["obstacleDetected"] = obstDetected;
  doc["obstacleSensor"]   = OBSTACLE_SENSOR;

  // DHT22
  if (dhtValid) {
    doc["temperature"] = cTemp;
    doc["humidity"]    = cHum;
    doc["heatIndex"]   = cHeatIdx;
  } else {
    doc["temperature"] = nullptr;
    doc["humidity"]    = nullptr;
    doc["heatIndex"]   = nullptr;
  }
  doc["dhtValid"] = dhtValid;

  // MPU6050_light — m/s² and deg/s (already converted by library)
  doc["accelX"]         = ax;
  doc["accelY"]         = ay;
  doc["accelZ"]         = az;
  doc["gyroX"]          = gx;
  doc["gyroY"]          = gy;
  doc["gyroZ"]          = gz;
  doc["pitch"]          = pitch;
  doc["roll"]           = roll;
  doc["accelMagnitude"] = aMag;

  // Rash driving flags
  doc["rashHarshBrake"] = rashBrake;
  doc["rashHarshAccel"] = rashAccel;
  doc["rashHarshTurn"]  = rashTurn;
  doc["rashBump"]       = rashBump;

  String data;
  serializeJson(doc, data);
  wsClient.sendTXT("42[\"driverLocation\"," + data + "]");

  // Clear rash flags after sending
  rashBrake = rashAccel = rashTurn = rashBump = false;

  Serial.printf("TX | Spd:%.1f Obs:%.0fcm T:%.1fC Ax:%.2f Ay:%.2f Az:%.2f Gz:%.1f\n",
    spd, obstDist,
    dhtValid ? cTemp : 0.0,
    ax, ay, az, gz);
}

// ─────────────────────────────────────────────────────────────────────
// SETUP
// ─────────────────────────────────────────────────────────────────────
void setup() {
  Serial.begin(115200);
  delay(500);
  Serial.println("\n╔═══════════════════════════╗");
  Serial.println(  "║   SmartBus Node v5.2       ║");
  Serial.println(  "║   MPU6050_light by rfetick ║");
  Serial.println(  "╚═══════════════════════════╝");

  // Obstacle sensor pins
  if (String(OBSTACLE_SENSOR) == "ULTRASONIC") {
    pinMode(TRIG_PIN, OUTPUT);
    pinMode(ECHO_PIN, INPUT);
    digitalWrite(TRIG_PIN, LOW);
    Serial.println("Obstacle: HC-SR04  TRIG=GPIO12  ECHO=GPIO14");
  } else {
    pinMode(IR_PIN, INPUT);
    Serial.println("Obstacle: IR sensor  OUT=GPIO15");
  }

  // I2C — shared by MPU6050 + OLED
  Wire.begin(I2C_SDA, I2C_SCL);
  Wire.setClock(400000); // 400kHz fast mode
  Serial.println("I2C: SDA=GPIO21  SCL=GPIO22  Speed=400kHz");

  // ── OLED ─────────────────────────────────────────────────────────
  if (!display.begin(SSD1306_SWITCHCAPVCC, OLED_ADDR)) {
    Serial.println("OLED: NOT FOUND at 0x3C — try 0x3D in OLED_ADDR");
    oledReady = false;
  } else {
    oledReady = true;
    display.clearDisplay();
    display.setTextSize(2);
    display.setTextColor(SSD1306_WHITE);
    display.setCursor(5, 5);  display.println(F("SmartBus"));
    display.setTextSize(1);
    display.setCursor(20, 30); display.println(F("v5.2 Starting"));
    display.setCursor(10, 46); display.print(F("Sensor: ")); display.println(OBSTACLE_SENSOR);
    display.display();
    Serial.println("OLED: Ready at 0x3C");
  }

  // ── MPU6050_light ─────────────────────────────────────────────────
  // begin() returns 0 on success
  byte mpuStatus = mpu.begin();
  if (mpuStatus != 0) {
    Serial.print("MPU6050: NOT FOUND — error code: ");
    Serial.println(mpuStatus);
    Serial.println("  Check: SDA=GPIO21, SCL=GPIO22, AD0=GND, VCC=3.3V");
    mpuReady = false;
  } else {
    mpuReady = true;
    Serial.println("MPU6050: Found! Calibrating — keep ESP32 STILL...");

    if (oledReady) {
      display.clearDisplay();
      display.setTextSize(1);
      display.setCursor(0, 20); display.println(F("MPU6050 Calibrating"));
      display.setCursor(0, 35); display.println(F("Keep device STILL!"));
      display.display();
    }

    // Calibrate offsets — device must be flat and still
    mpu.calcOffsets();
    Serial.println("MPU6050: Calibration done!");
  }

  // ── DHT22 ─────────────────────────────────────────────────────────
  dht.begin();
  Serial.println("DHT22: GPIO4 initialized");

  if (oledReady) {
    display.clearDisplay();
    display.setTextSize(1);
    display.setCursor(0, 25); display.println(F("Waiting for DHT22..."));
    display.display();
  }
  delay(2000); // DHT22 warmup time
  readDHT();
  Serial.println(dhtValid ? "DHT22: OK" : "DHT22: FAILED — check GPIO4");

  // ── GPS ───────────────────────────────────────────────────────────
  gpsSerial.begin(9600, SERIAL_8N1, GPS_RX_PIN, GPS_TX_PIN);
  Serial.println("GPS: UART2  RX=GPIO16  TX=GPIO17");

  // ── WiFi ──────────────────────────────────────────────────────────
  if (oledReady) {
    display.clearDisplay();
    display.setTextSize(1);
    display.setCursor(0, 10); display.println(F("Connecting to WiFi:"));
    display.setCursor(0, 25); display.println(WIFI_SSID);
    display.setCursor(0, 42); display.println(F("Please wait..."));
    display.display();
  }

  Serial.print("WiFi: Connecting to "); Serial.println(WIFI_SSID);
  WiFi.begin(WIFI_SSID, WIFI_PASSWORD);
  int tries = 0;
  while (WiFi.status() != WL_CONNECTED && tries < 40) {
    delay(500); Serial.print("."); tries++;
  }

  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("\nWiFi: " + WiFi.localIP().toString());
    if (oledReady) {
      display.clearDisplay();
      display.setTextSize(1);
      display.setCursor(5, 10);  display.println(F("WiFi Connected!"));
      display.setCursor(0, 28);  display.println(WiFi.localIP().toString());
      display.setCursor(0, 46);  display.println(F("Connecting server..."));
      display.display();
      delay(1500);
    }
  } else {
    Serial.println("\nWiFi: FAILED — check SSID/password");
    if (oledReady) {
      display.clearDisplay();
      display.setCursor(15, 20); display.println(F("WiFi FAILED!"));
      display.setCursor(0, 38);  display.println(F("Check credentials"));
      display.display();
    }
  }

  // ── WebSocket ─────────────────────────────────────────────────────
  wsClient.begin(SERVER_HOST, SERVER_PORT,
                 "/socket.io/?EIO=4&transport=websocket");
  wsClient.onEvent(webSocketEvent);
  wsClient.setReconnectInterval(T_RECONNECT);
  wsClient.setExtraHeaders(
    ("Authorization: Bearer " + String(JWT_TOKEN)).c_str());

  Serial.println("\n╔═══════════ READY ═══════════╗");
  Serial.printf( "║  Role:    %-19s║\n", DEVICE_ROLE);
  Serial.printf( "║  Sensor:  %-19s║\n", OBSTACLE_SENSOR);
  Serial.printf( "║  OLED:    %-19s║\n", oledReady ? "OK (0x3C)" : "NOT FOUND");
  Serial.printf( "║  MPU6050: %-19s║\n", mpuReady  ? "OK (calibrated)" : "NOT FOUND");
  Serial.printf( "║  DHT22:   %-19s║\n", dhtValid  ? "OK" : "CHECK GPIO4");
  Serial.println("╚════════════════════════════╝");
}

// ─────────────────────────────────────────────────────────────────────
// MAIN LOOP
// ─────────────────────────────────────────────────────────────────────
void loop() {
  wsClient.loop();

  // Feed GPS parser
  while (gpsSerial.available()) gps.encode(gpsSerial.read());

  unsigned long now = millis();

  // MPU6050 — every 100ms (high freq for rash detection)
  if (now - tMPU >= T_MPU) {
    tMPU = now;
    sampleMPU();
  }

  // DHT22 — every 3s
  if (now - tDHT >= T_DHT) {
    tDHT = now;
    readDHT();
  }

  // OLED — every 1s
  if (now - tOLED >= T_OLED) {
    tOLED = now;
    readObstacle();
    updateOLED();
  }

  // Telemetry — every 2s
  if (now - tTelemetry >= T_TELEMETRY) {
    tTelemetry = now;
    readObstacle();
    emitTelemetry();
  }
}
